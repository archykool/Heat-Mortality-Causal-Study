import pandas as pd
import numpy as np
from scipy import stats
from sklearn.preprocessing import StandardScaler
import statsmodels.api as sm
import matplotlib.pyplot as plt
import seaborn as sns
import warnings
warnings.filterwarnings('ignore')

# Load data
print("Loading data...")
data = pd.read_csv("merged_data.csv")  # Replace with your data path
print(f"Data loaded: {data.shape[0]} rows and {data.shape[1]} columns\n")

# Define variables
treatment_col = 'Heat Wave Days Based on Daily Maximum Temperature'
outcome_col = 'Deaths_x'
confounders = [
    '% Smokers',
    '% Binge Drinking',
    'No of Diabetics',
    'PM Days',
    '% Healthy Food',
    '% unemployed'
]

# Preprocess data
print("Preprocessing data...")
relevant_cols = confounders + [treatment_col, outcome_col]
processed_data = data[relevant_cols].copy()

# Handle missing values and infinities
numeric_cols = processed_data.select_dtypes(include=[np.number]).columns
for col in numeric_cols:
    processed_data[col].fillna(processed_data[col].median(), inplace=True)
processed_data = processed_data.replace([np.inf, -np.inf], np.nan)
processed_data = processed_data.fillna(processed_data.median())

print("Step 1: Running basic regression as benchmark...")
# Basic linear regression
linear_model = sm.OLS(processed_data[outcome_col], 
                     sm.add_constant(processed_data[treatment_col])).fit()

# Basic polynomial regression
data_poly = processed_data.copy()
data_poly['treatment_squared'] = processed_data[treatment_col]**2
data_poly['treatment_cubic'] = processed_data[treatment_col]**3

poly_model = sm.OLS(
    data_poly[outcome_col],
    sm.add_constant(data_poly[[treatment_col, 'treatment_squared', 'treatment_cubic']])
).fit()

print("\nBasic Linear Regression Results:")
print("-" * 50)
print(linear_model.summary().tables[1])

print("\nBasic Polynomial Regression Results:")
print("-" * 50)
print(poly_model.summary().tables[1])

print("\nStep 2: Estimating GPS...")
# GPS estimation
X = processed_data[confounders]
T = processed_data[treatment_col]

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

gps_model = sm.OLS(T, sm.add_constant(X_scaled)).fit()
pred_t = gps_model.predict(sm.add_constant(X_scaled))
sigma = np.std(T - pred_t)
gps = stats.norm.pdf(T, pred_t, sigma)

print("\nGPS Model Results:")
print("-" * 50)
print(gps_model.summary().tables[1])

print("\nStep 3: Calculating IPW weights...")
# Calculate weights
T_std = (T - T.mean()) / T.std()
numerator = stats.norm.pdf(T_std, 0, 1)
denominator = gps + 1e-8
weights = numerator / denominator
weights = np.clip(weights, np.percentile(weights, 1), np.percentile(weights, 99))
weights = weights * (len(weights) / weights.sum())

print("\nStep 4: Checking balance...")
# Balance checking
balance_stats_list = []
for confounder in confounders:
    before_corr = stats.pearsonr(processed_data[treatment_col], processed_data[confounder])[0]
    weighted_corr = np.cov(processed_data[treatment_col], processed_data[confounder], 
                          aweights=weights)[0,1] / (np.std(processed_data[treatment_col]) * np.std(processed_data[confounder]))
    
    balance_stats_list.append({
        'Confounder': confounder,
        'Before_Correlation': before_corr,
        'After_Correlation': weighted_corr
    })

balance_stats = pd.DataFrame(balance_stats_list)
print("\nBalance Assessment:")
print("-" * 50)
for _, row in balance_stats.iterrows():
    print(f"\n{row['Confounder']}:")
    print(f"Before weighting: {row['Before_Correlation']:.3f}")
    print(f"After weighting: {row['After_Correlation']:.3f}")

print("\nStep 5: Estimating weighted treatment effects...")
# Weighted regression
weighted_model = sm.WLS(processed_data[outcome_col], 
                       sm.add_constant(processed_data[treatment_col]),
                       weights=weights).fit()

weighted_poly_model = sm.WLS(
    data_poly[outcome_col],
    sm.add_constant(data_poly[[treatment_col, 'treatment_squared', 'treatment_cubic']]),
    weights=weights
).fit()

print("\nWeighted Linear Regression Results:")
print("-" * 50)
print(weighted_model.summary().tables[1])

print("\nWeighted Polynomial Regression Results:")
print("-" * 50)
print(weighted_poly_model.summary().tables[1])

# Create visualizations
plt.figure(figsize=(15, 5))

# Plot 1: Basic vs Weighted Linear
plt.subplot(1, 3, 1)
plt.scatter(processed_data[treatment_col], processed_data[outcome_col], alpha=0.1)
x_range = np.linspace(processed_data[treatment_col].min(), processed_data[treatment_col].max(), 100)
plt.plot(x_range, linear_model.params[0] + linear_model.params[1] * x_range, 'r-', label='Basic Linear')
plt.plot(x_range, weighted_model.params[0] + weighted_model.params[1] * x_range, 'b-', label='Weighted Linear')
plt.xlabel('Heat Wave Days')
plt.ylabel('Deaths')
plt.title('Linear Models Comparison')
plt.legend()

# Plot 2: Basic vs Weighted Polynomial
plt.subplot(1, 3, 2)
plt.scatter(processed_data[treatment_col], processed_data[outcome_col], alpha=0.1)
y_basic = (poly_model.params[0] + poly_model.params[1] * x_range +
           poly_model.params[2] * x_range**2 + poly_model.params[3] * x_range**3)
y_weighted = (weighted_poly_model.params[0] + weighted_poly_model.params[1] * x_range +
              weighted_poly_model.params[2] * x_range**2 + weighted_poly_model.params[3] * x_range**3)
plt.plot(x_range, y_basic, 'r-', label='Basic Polynomial')
plt.plot(x_range, y_weighted, 'b-', label='Weighted Polynomial')
plt.xlabel('Heat Wave Days')
plt.ylabel('Deaths')
plt.title('Polynomial Models Comparison')
plt.legend()

# Plot 3: Balance Comparison
plt.subplot(1, 3, 3)
x = np.arange(len(confounders))
width = 0.35
plt.bar(x - width/2, balance_stats['Before_Correlation'], width, label='Before GPS')
plt.bar(x + width/2, balance_stats['After_Correlation'], width, label='After GPS')
plt.axhline(y=0, color='k', linestyle='-', alpha=0.2)
plt.axhline(y=0.1, color='r', linestyle='--', alpha=0.2)
plt.axhline(y=-0.1, color='r', linestyle='--', alpha=0.2)
plt.xlabel('Confounders')
plt.ylabel('Correlation with Treatment')
plt.title('Balance Comparison')
plt.xticks(x, balance_stats['Confounder'], rotation=45, ha='right')
plt.legend()

plt.tight_layout()
plt.show()

print("\nAnalysis complete!")
